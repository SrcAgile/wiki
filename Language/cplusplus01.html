开始

  1. code
#include <iostream>
using std::cout;
namespace nameA{
    void sayhello(){
        cout<<"hello i am nameA";
    }
}
namespace nameB{
    void sayhello(){
        cout<<"hello i am nameB";
    }
}
using namespace nameB;
int main(){
    sayhello();
    return 0;
}
